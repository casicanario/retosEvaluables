┌─ Query Tab ─────────────────────────────────────────────────────┐
│                                                                 │
│ SELECT                                                          │
│     s.subject_id,                                              │
│     s.title as asignatura,                                     │
│     ROUND(AVG(m.mark), 2) as media_ultimo_ano,                │
│     COUNT(m.mark_id) as total_examenes                         │
│ FROM subjects s                                                 │
│ INNER JOIN marks m ON s.subject_id = m.subject_id             │
│ WHERE YEAR(m.date) = 2024                                      │
│   AND s.subject_id = 1                                         │
│ GROUP BY s.subject_id, s.title;                               │
│                                                                 │
